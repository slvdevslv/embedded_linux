Below is a **single self-contained `.c` file** that demonstrates **Linux system programming concepts in increasing complexity**.

Design goals:

* Starts with **basic system calls**
* Moves to **file operations**
* Then **process management**
* Then **IPC (pipes)**
* Then **signals**
* Then **memory mapping**
* Then **shared memory**
* Includes **clear explanations in comments**
* Uses a **menu-driven `main()`** so the whole file compiles as one program

You can compile it with:

```bash
gcc linux_sysprog_learning.c -o linux_sysprog_learning
```

Run:

```bash
./linux_sysprog_learning
```

---

```c
/*
    linux_sysprog_learning.c

    A progressive set of Linux System Programming demonstrations.
    Concepts covered:

    1. write() system call
    2. open(), read(), write(), close()
    3. lseek()
    4. stat()
    5. fork()
    6. fork() + exec()
    7. wait()
    8. pipe() IPC
    9. dup2() for redirection
    10. signals
    11. mmap()
    12. POSIX shared memory

    Compile:
        gcc linux_sysprog_learning.c -o linux_sysprog_learning -lrt

    Run:
        ./linux_sysprog_learning
*/

#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <fcntl.h>
#include <string.h>
#include <sys/types.h>
#include <sys/stat.h>
#include <sys/wait.h>
#include <signal.h>
#include <sys/mman.h>

/* --------------------------------------------------------- */
/* 1. Basic write() system call                              */
/* --------------------------------------------------------- */

void demo_write_syscall()
{
    const char *msg = "Hello using write() system call\n";

    /* write(fd, buffer, size) */
    write(STDOUT_FILENO, msg, strlen(msg));
}

/* --------------------------------------------------------- */
/* 2. File creation and writing using open/write             */
/* --------------------------------------------------------- */

void demo_file_write()
{
    int fd;

    fd = open("demo_file.txt", O_CREAT | O_WRONLY | O_TRUNC, 0644);
    if (fd < 0)
    {
        perror("open");
        return;
    }

    const char *data = "Linux System Programming\n";

    write(fd, data, strlen(data));

    close(fd);

    printf("File written successfully\n");
}

/* --------------------------------------------------------- */
/* 3. Reading from a file                                    */
/* --------------------------------------------------------- */

void demo_file_read()
{
    char buffer[100];
    int fd = open("demo_file.txt", O_RDONLY);

    if (fd < 0)
    {
        perror("open");
        return;
    }

    int bytes = read(fd, buffer, sizeof(buffer)-1);

    if (bytes < 0)
    {
        perror("read");
        close(fd);
        return;
    }

    buffer[bytes] = '\0';

    printf("Read from file:\n%s\n", buffer);

    close(fd);
}

/* --------------------------------------------------------- */
/* 4. lseek() demonstration                                  */
/* --------------------------------------------------------- */

void demo_lseek()
{
    int fd = open("demo_file.txt", O_RDONLY);

    if (fd < 0)
    {
        perror("open");
        return;
    }

    lseek(fd, 6, SEEK_SET);

    char buffer[50];

    int n = read(fd, buffer, sizeof(buffer)-1);

    buffer[n] = '\0';

    printf("After seeking 6 bytes:\n%s\n", buffer);

    close(fd);
}

/* --------------------------------------------------------- */
/* 5. stat() system call                                     */
/* --------------------------------------------------------- */

void demo_stat()
{
    struct stat fileStat;

    if (stat("demo_file.txt", &fileStat) < 0)
    {
        perror("stat");
        return;
    }

    printf("File size: %ld bytes\n", fileStat.st_size);
    printf("Number of links: %ld\n", fileStat.st_nlink);
    printf("Inode: %ld\n", fileStat.st_ino);
}

/* --------------------------------------------------------- */
/* 6. fork() demonstration                                   */
/* --------------------------------------------------------- */

void demo_fork()
{
    pid_t pid = fork();

    if (pid < 0)
    {
        perror("fork");
        return;
    }

    if (pid == 0)
    {
        printf("Child process PID: %d\n", getpid());
    }
    else
    {
        printf("Parent process PID: %d\n", getpid());
        printf("Child PID returned to parent: %d\n", pid);
    }
}

/* --------------------------------------------------------- */
/* 7. fork() + exec() demonstration                          */
/* --------------------------------------------------------- */

void demo_exec()
{
    pid_t pid = fork();

    if (pid == 0)
    {
        printf("Executing ls command\n");

        execl("/bin/ls", "ls", "-l", NULL);

        perror("exec failed");
        exit(1);
    }
    else
    {
        wait(NULL);
        printf("Child finished execution\n");
    }
}

/* --------------------------------------------------------- */
/* 8. pipe() IPC                                             */
/* --------------------------------------------------------- */

void demo_pipe()
{
    int fd[2];
    pipe(fd);

    pid_t pid = fork();

    if (pid == 0)
    {
        close(fd[1]);

        char buffer[50];

        read(fd[0], buffer, sizeof(buffer));

        printf("Child received: %s\n", buffer);

        close(fd[0]);
    }
    else
    {
        close(fd[0]);

        char msg[] = "Hello from parent via pipe";

        write(fd[1], msg, strlen(msg)+1);

        close(fd[1]);

        wait(NULL);
    }
}

/* --------------------------------------------------------- */
/* 9. dup2() redirection                                     */
/* --------------------------------------------------------- */

void demo_dup2()
{
    int fd = open("redirect_output.txt", O_CREAT | O_WRONLY | O_TRUNC, 0644);

    if (fd < 0)
    {
        perror("open");
        return;
    }

    dup2(fd, STDOUT_FILENO);

    printf("This line goes into the file!\n");

    close(fd);
}

/* --------------------------------------------------------- */
/* 10. Signal handling                                       */
/* --------------------------------------------------------- */

void signal_handler(int sig)
{
    printf("Received signal: %d\n", sig);
}

void demo_signal()
{
    signal(SIGINT, signal_handler);

    printf("Press Ctrl+C to trigger SIGINT\n");

    while(1)
        sleep(1);
}

/* --------------------------------------------------------- */
/* 11. mmap() demonstration                                  */
/* --------------------------------------------------------- */

void demo_mmap()
{
    int fd = open("demo_file.txt", O_RDONLY);

    struct stat sb;

    fstat(fd, &sb);

    char *mapped = mmap(NULL, sb.st_size, PROT_READ, MAP_PRIVATE, fd, 0);

    if (mapped == MAP_FAILED)
    {
        perror("mmap");
        close(fd);
        return;
    }

    write(STDOUT_FILENO, mapped, sb.st_size);

    munmap(mapped, sb.st_size);

    close(fd);
}

/* --------------------------------------------------------- */
/* 12. POSIX shared memory                                   */
/* --------------------------------------------------------- */

void demo_shared_memory()
{
    const char *name = "/myshm";

    int fd = shm_open(name, O_CREAT | O_RDWR, 0666);

    ftruncate(fd, 4096);

    char *ptr = mmap(0, 4096, PROT_WRITE, MAP_SHARED, fd, 0);

    strcpy(ptr, "Hello from shared memory");

    printf("Data written to shared memory\n");

    munmap(ptr, 4096);

    close(fd);

    shm_unlink(name);
}

/* --------------------------------------------------------- */
/* Menu                                                      */
/* --------------------------------------------------------- */

void print_menu()
{
    printf("\nLinux System Programming Learning\n");
    printf("---------------------------------\n");

    printf("1  - write() syscall\n");
    printf("2  - File write\n");
    printf("3  - File read\n");
    printf("4  - lseek()\n");
    printf("5  - stat()\n");
    printf("6  - fork()\n");
    printf("7  - exec()\n");
    printf("8  - pipe()\n");
    printf("9  - dup2()\n");
    printf("10 - signals\n");
    printf("11 - mmap()\n");
    printf("12 - shared memory\n");
    printf("0  - exit\n");
}

/* --------------------------------------------------------- */
/* Main                                                      */
/* --------------------------------------------------------- */

int main()
{
    int choice;

    while (1)
    {
        print_menu();

        printf("Enter choice: ");
        scanf("%d", &choice);

        switch(choice)
        {
            case 1: demo_write_syscall(); break;
            case 2: demo_file_write(); break;
            case 3: demo_file_read(); break;
            case 4: demo_lseek(); break;
            case 5: demo_stat(); break;
            case 6: demo_fork(); break;
            case 7: demo_exec(); break;
            case 8: demo_pipe(); break;
            case 9: demo_dup2(); break;
            case 10: demo_signal(); break;
            case 11: demo_mmap(); break;
            case 12: demo_shared_memory(); break;
            case 0: exit(0);
            default: printf("Invalid choice\n");
        }
    }

    return 0;
}
```

## Concepts Covered in the Correct Learning Order

This file teaches Linux system programming in a **natural progression**:

1. **Basic syscall**

   * `write()`

2. **File system interaction**

   * `open()`
   * `read()`
   * `write()`
   * `close()`

3. **File positioning**

   * `lseek()`

4. **File metadata**

   * `stat()`

5. **Process management**

   * `fork()`

6. **Program execution**

   * `exec()`

7. **Process synchronization**

   * `wait()`

8. **Interprocess Communication**

   * `pipe()`

9. **File descriptor manipulation**

   * `dup2()`

10. **Signals**

* `signal()`

11. **Memory mapping**

* `mmap()`

12. **Shared memory IPC**

* `shm_open()`
